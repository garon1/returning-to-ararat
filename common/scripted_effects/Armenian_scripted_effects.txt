ARM_increase_revolt_risk = {
	custom_effect_tooltip = ARM_increase_revolt_risk_tt
	hidden_effect = {
		add_to_variable = { ARM_revolt_risk = 1 }
		clamp_variable = {
			var = ARM_revolt_risk
			min = 0
			max = 5
		}
		if = {
			limit = {
				has_resistance = no
				ARM_is_turkish_state = no
				check_variable = { ARM_revolt_risk > 2 }
				is_core_of = ARM
			}
			force_enable_resistance = {
				occupier = ARM
			}
			start_resistance = yes
			set_resistance = 10
		}
		if = {
			limit = {
				check_variable = { ARM_revolt_risk = 5 }
			}
			if = {
				limit = {
					has_dynamic_modifier = { modifier = high_revolt_risk }
				}
				remove_dynamic_modifier = { modifier = high_revolt_risk }
			}
			add_dynamic_modifier = { modifier = very_high_revolt_risk }
		}
		else_if = {
			limit = {
				check_variable = { ARM_revolt_risk = 4 }
			}
			add_dynamic_modifier = { modifier = high_revolt_risk }
		}
		ARM_evaluate_stabillity_modifier = yes
	}
}

ARM_evaluate_stabillity_modifier = {
	ARM = {
		set_variable = { var_state_stability = 0 }
		set_variable = { var_state_martial_law = 0 }
		hidden_effect = {
			every_owned_state = {
				limit = { ARM_revolt_risk_low = yes }
				if = {
					limit = {
						check_variable = { ARM_revolt_risk = 5 }
					}
					add_to_variable = { ARM.var_state_stability = -0.05 }
				}
				if = {
					limit = {
						check_variable = { ARM_revolt_risk = 4 }
					}
					add_to_variable = { ARM.var_state_stability = -0.03 }
				}
				else_if = {
					limit = {
						check_variable = { ARM_revolt_risk = 3 }
					}
					add_to_variable = { ARM.var_state_stability = -0.02 }
				}
				else_if = {
					limit = {
						check_variable = { ARM_revolt_risk = 2 }
					}
					add_to_variable = { ARM.var_state_stability = -0.01 }
				}
			}
		}
		force_update_dynamic_modifier = yes
	}
}

	}
}
