USA_low_legionist_support = {
	custom_effect_tooltip = USA_low_legionist_support
	hidden_effect = {
		if = {
			limit = { is_owned_by = USA }
			if = {
				limit = {
					has_dynamic_modifier = { modifier = low_legionist_support }
				}
				remove_dynamic_modifier = { modifier = low_legionist_support }
			}
			if = {
				limit = {
					NOT = { has_dynamic_modifier = { modifier = low_legionist_support } }
				}
				add_dynamic_modifier = { modifier = low_legionist_support }
			}
		}
	}
}
